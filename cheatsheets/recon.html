01Passive Reconnaissance
Passive recon = no direct interaction with the target. You gather information from public sources, search engines, and third-party databases. Zero chance of detection.
# WHOIS & Domain Info
bash
# WHOIS lookup
whois target.com

# DNS lookup
nslookup target.com
dig target.com
dig target.com ANY              # all records
dig target.com MX               # mail records
dig target.com TXT              # text records (SPF, DKIM)
dig target.com NS               # name servers
host target.com
# Certificate Transparency — Free Subdomain Intel
bash
# crt.sh — SSL cert logs (browser or curl)
curl -s "https://crt.sh/?q=%25.target.com&output=json" | jq -r '.[].name_value' | sort -u

# Also check manually at:
https://crt.sh/?q=%.target.com
https://transparencyreport.google.com/https/certificates
# Wayback Machine
bash
# Get all archived URLs
curl "http://web.archive.org/cdx/search/cdx?url=*.target.com&output=text&fl=original&collapse=urlkey"

# gau — fetch known URLs
gau target.com
gau target.com | grep ".php"     # filter PHP pages

03DNS Enumeration
# Zone Transfer (AXFR)
bash
# Find nameservers first
dig NS target.com

# Attempt zone transfer
dig axfr @ns1.target.com target.com
host -t axfr target.com ns1.target.com

# dnsenum — full DNS enum + zone transfer attempt
dnsenum target.com
dnsenum --dnsserver ns1.target.com target.com
# DNS Brute Force
bash
# dnsrecon
dnsrecon -d target.com -D /usr/share/wordlists/dnsmap.txt -t brt

# fierce
fierce --domain target.com

# Reverse DNS lookup
dnsrecon -r 192.168.1.0/24   # reverse lookup on range
04Subdomain Enumeration
# Amass — Comprehensive Subdomain Enum
bash
# Passive (no direct contact)
amass enum -passive -d target.com

# Active (DNS brute force)
amass enum -active -d target.com

# Save output
amass enum -d target.com -o subdomains.txt
# Subfinder & Gobuster DNS
bash
# subfinder
subfinder -d target.com
subfinder -d target.com -o subs.txt

# gobuster DNS mode
gobuster dns -d target.com -w /usr/share/seclists/Discovery/DNS/subdomains-top1million-5000.txt

# ffuf DNS
ffuf -u http://FUZZ.target.com -w /usr/share/seclists/Discovery/DNS/subdomains-top1million-5000.txt -fc 302
# VHost Fuzzing (Virtual Host Discovery)
bash
# ffuf vhost fuzzing (when multiple sites on same IP)
ffuf -u http://TARGET_IP -H "Host: FUZZ.target.com" -w subdomains.txt -fs 4242
# -fs filters by response size (adjust to filter default page size)
05Directory & File Scanning
# ffuf — Fast Fuzzing
bash
# Basic directory scan
ffuf -u http://target.com/FUZZ -w /usr/share/seclists/Discovery/Web-Content/common.txt

# With extensions
ffuf -u http://target.com/FUZZ -w wordlist.txt -e .php,.html,.txt,.bak,.zip,.old

# Filter by status code
ffuf -u http://target.com/FUZZ -w wordlist.txt -mc 200,301,302,403

# Filter by response size (remove false positives)
ffuf -u http://target.com/FUZZ -w wordlist.txt -fs 1234

# Recursive scan
ffuf -u http://target.com/FUZZ -w wordlist.txt -recursion -recursion-depth 2

# POST parameter fuzzing
ffuf -u http://target.com/login.php -X POST -d "username=FUZZ&password=test" -w users.txt
# Gobuster
bash
# Directory scan
gobuster dir -u http://target.com -w /usr/share/wordlists/dirbuster/directory-list-2.3-medium.txt -x php,html,txt

# With auth
gobuster dir -u http://target.com -w wordlist.txt -U admin -P password

# Quiet mode (results only)
gobuster dir -u http://target.com -w wordlist.txt -q
# Dirsearch
bash
dirsearch -u http://target.com
dirsearch -u http://target.com -e php,html,txt
dirsearch -u http://target.com -w /usr/share/wordlists/custom.txt
06Active Scanning
Active recon = direct interaction with target. Leaves logs. Only do this during authorized engagements within defined scope.
# Nmap — Service & Version Detection
bash
# Full recon scan
nmap -sC -sV -A -p- TARGET_IP -oN nmap_full.txt

# HTTP enumeration script
nmap --script http-enum -p 80 TARGET_IP

# HTTP methods
nmap --script http-methods -p 80 TARGET_IP

# HTTP title scan (find web apps on all ports)
nmap --script http-title -sV -p 80,443,8080,8443,8000 192.168.1.0/24
# WAF Detection
bash
# wafw00f — detect Web Application Firewall
wafw00f http://target.com
wafw00f -a http://target.com   # check all WAFs
07Google Dorks
# Google Dork Operators
dorks
# Find subdomains
site:*.target.com

# Exposed sensitive files
site:target.com ext:pdf OR ext:doc OR ext:xls OR ext:txt
site:target.com ext:sql OR ext:bak OR ext:log

# Login pages
site:target.com inurl:login OR inurl:admin OR inurl:portal

# Config files
site:target.com inurl:config OR inurl:env OR inurl:settings

# Exposed credentials in code (GitHub)
site:github.com "target.com" password
site:github.com "target.com" api_key

# Directory listings
site:target.com intitle:"index of"

# Cameras / exposed devices
inurl:"viewerframe?mode=motion"
intitle:"webcamXP 5"

# Error pages (reveals tech stack)
site:target.com "ORA-" OR "SQL Server" OR "php error"
