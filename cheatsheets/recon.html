🌐 DNS Enumeration

DNS Lookups 
# Basic DNS records
dig target.com A
dig target.com MX
dig target.com NS
dig target.com ANY

# Reverse lookup
dig -x 192.168.1.100

# Zone transfer attempt
dig axfr target.com @ns1.target.com

# nslookup interactive
nslookup
> server ns1.target.com
> set type=any
> target.com

🗺️ Port Scanning

Nmap Recon Workflow 
# Step 1 — Quick top ports scan
nmap -sC -sV --top-ports 1000 -oN quick.txt target.com

# Step 2 — Full port scan (all 65535)
nmap -p- --min-rate 5000 -oN allports.txt target.com

# Step 3 — Targeted scan on discovered ports
nmap -sC -sV -p 22,80,443,8080 -oN targeted.txt target.com

# UDP top 100
nmap -sU --top-ports 100 -oN udp.txt target.com

📁 Directory Busting

ffuf & Dirsearch 
# ffuf — fast, versatile
ffuf -u http://target.com/FUZZ -w /usr/share/seclists/Discovery/Web-Content/common.txt -mc 200,301,302,403

# ffuf with extensions
ffuf -u http://target.com/FUZZ -w /usr/share/seclists/Discovery/Web-Content/raft-medium-words.txt -e .php,.txt,.html,.bak

# Dirsearch
dirsearch -u http://target.com -e php,html,js,txt -t 50

# Gobuster
gobuster dir -u http://target.com -w /usr/share/wordlists/dirb/common.txt -x php,html -t 50

🔗 Subdomain Enumeration

Subdomain Discovery 
# ffuf virtual host fuzzing
ffuf -u http://target.com -H "Host: FUZZ.target.com" -w /usr/share/seclists/Discovery/DNS/subdomains-top1million-5000.txt -mc 200

# DNS brute force with ffuf
ffuf -u http://FUZZ.target.com -w /usr/share/seclists/Discovery/DNS/subdomains-top1million-5000.txt -mc 200
